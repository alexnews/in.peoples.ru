CREATE TABLE `event` (
  `event_id` int NOT NULL AUTO_INCREMENT,
  `date_calendar` date NOT NULL,
  `event_txt` text,
  `do_n_ery` int NOT NULL DEFAULT '0' -- Before the Common Era if 1 it means before 0,
  `source` varchar(15) DEFAULT NULL,
  `date_registration` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `year` int DEFAULT NULL,
  `month` int DEFAULT NULL,
  `day` int DEFAULT NULL,
  PRIMARY KEY (`event_id`),
  KEY `calendar_event_year_idx` (`year`),
  KEY `calendar_event_month_idx` (`month`),
  KEY `calendar_event_day_idx` (`day`)
) ENGINE=MyISAM AUTO_INCREMENT=27783 DEFAULT CHARSET=cp1251;

CREATE TABLE `event_photo` (
  `event_photo_id` int NOT NULL AUTO_INCREMENT,
  `event_id` int NOT NULL,
  `photoname` varchar(50) CHARACTER SET cp1251 COLLATE cp1251_general_ci NOT NULL,
  `photodescription` varchar(250) DEFAULT NULL,
  `source` varchar(100) DEFAULT NULL,
  `date_registration` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`event_photo_id`),
  KEY `calendar_event_id_idx` (`event_id`)
) ENGINE=MyISAM AUTO_INCREMENT=1261 DEFAULT CHARSET=cp1251;

CREATE TABLE `persons` (
  `Persons_id` int NOT NULL AUTO_INCREMENT,
  `NameRus` varchar(40) DEFAULT NULL,
  `SurNameRus` varchar(40) DEFAULT NULL,
  `2NameRus` varchar(40) DEFAULT NULL,
  `FullNameRus` varchar(60) DEFAULT NULL,
  `NameEngl` varchar(40) DEFAULT NULL,
  `SurNameEngl` varchar(40) DEFAULT NULL,
  `2NameEngl` varchar(40) DEFAULT NULL,
  `FullNameEngl` varchar(60) DEFAULT NULL,
  `AllUrlInSity` varchar(100) DEFAULT '',
  `NamePhoto` varchar(50) DEFAULT NULL,
  `DateIn` date DEFAULT NULL,
  `DateOut` date DEFAULT NULL,
  `popularity` enum('1','2','3','4','5') DEFAULT '1',
  `date_registration` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `date_updated` timestamp NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`Persons_id`),
  UNIQUE KEY `AllUrlInSity` (`AllUrlInSity`)
) ENGINE=InnoDB AUTO_INCREMENT=125023 DEFAULT CHARSET=cp1251;
